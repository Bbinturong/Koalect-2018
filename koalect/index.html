Exercice 1 Temps approximatif : 30min

Créer une fonction qui permette de supprimer tous les caractères doubles d’une chaîne de caractère donnée. Ex: « Je suis un homme » doit devenir « Jinho »

Réponse : 

                    
                       
        function removeDuplicate() {

            var exo1 = $('#koalect-js-1'); 
            var array = exo1.val().toLowerCase().split('');            
            var indexArray = []; 

            $.each(array, function(i) { 

                var test = $.inArray(array[i], array);
                if (test != i) {

                    indexArray.push(test, i);
                }
            });

            $.each(indexArray, function(i) {

                delete array[indexArray[i]];
            })

            array = array.filter(Boolean);
            array[0] = array[0].toUpperCase();
            $('#koalect-js-1-answer').html(array);

        }

Exercice 2 Temps approximatif : 20min

Le palindrome est un mot qui se lit dans les deux sens. Créer une fonction qui permette de vérifier si oui ou non, la chaine de caractère donnée est un palindrome. Ex: "kayak" = true et "hello" = false

Réponse : 

                    
                       
        function checkPalindrome() {

            var exo2 = $('#koalect-js-2');
            var array = exo2.val().toLowerCase().split('');

            var palindrome = true;

            $.each(array, function(i) {

                var j = array.length - 1 - i;
                if (array[i] != array[j]) {
                    palindrome = false;
                    return false;
                }

            })

            if (palindrome) {
                $('#koalect-js-2-answer').html('C est un palindrome');
            } else {
                $('#koalect-js-2-answer').html('C est pas un palindrome');
            }


        }

Exercice 3 Temps approximatif : 20min

À partir de deux arrays différentes, trouver une fonction qui permette de récupérer tous les nombres communs aux deux arrays. Ex: [2, 4, 5, 8, 9] et [1, 4, 5, 9, 10, 11] = [4, 5, 9]

Réponse : 

                    
          function checkSameNumbers() {

            var exo3_1 = $('#koalect-js-3-1');
            var exo3_2 = $('#koalect-js-3-2');
            var array1 = exo3_1.val().split('');
            var array2 = exo3_2.val().split('');

            var uniqueArray = [];
            $.each(array1, function(i) {

                if ($.inArray(array1[i], array2) >= 0 &&
                    $.inArray(array1[i], uniqueArray)) {
                    uniqueArray.push(array1[i]);
                }

            });

            $('#koalect-js-3-answer').html(uniqueArray);
        }
